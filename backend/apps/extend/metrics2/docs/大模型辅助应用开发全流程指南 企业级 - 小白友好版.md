### 2.2 模型客户端 (core/model_client.py)

```
import openai
import time
import asyncio
from typing import Dict, Any, Optional
from core.workflow import WorkflowStep
from utils.logger import get_logger
from config.settings import settings

logger = get_logger(__name__)

class ModelClient:
    def __init__(self):
        openai.api_key = settings.OPENAI_API_KEY
        openai.api_base = settings.OPENAI_API_BASE
        
    async def generate(
        self,
        prompt: str,
        max_tokens: Optional[int] = None,
        temperature: float = 0.7,
        retry_count: int = 0
    ) -> Dict[str, Any]:
        """
        异步生成模型响应
        """
        try:
            response = await openai.ChatCompletion.acreate(
                model=settings.MODEL_NAME,
                messages=[{"role": "user", "content": prompt}],
                max_tokens=max_tokens or settings.MAX_TOKENS,
                temperature=temperature,
                timeout=settings.REQUEST_TIMEOUT
            )
            
            return {
                "success": True,
                "content": response.choices[0].message.content,
                "usage": response.usage
            }
            
        except Exception as e:
            logger.error(f"Model request failed: {str(e)}")
            
            # 指数退避重试
            if retry_count < settings.MAX_RETRIES:
                wait_time = 2 ** retry_count
                await asyncio.sleep(wait_time)
                return await self.generate(prompt, max_tokens, temperature, retry_count + 1)
            else:
                return {
                    "success": False,
                    "error": str(e),
                    "content": None
                }
    
    def sync_generate(self, *args, **kwargs) -> Dict[str, Any]:
        """同步版本的generate方法"""
        return asyncio.run(self.generate(*args, **kwargs))
```

### 2.3 Agent基类 (core/agent_base.py)

```
from abc import ABC, abstractmethod
from typing import Dict, Any
from utils.logger import get_logger

logger = get_logger(__name__)

class BaseAgent(ABC):
    def __init__(self, name: str):
        self.name = name
        self.logger = logger
    
    @abstractmethod
    async def execute(self, context: Dict[str, Any]) -> Dict[str, Any]:
        """执行Agent任务"""
        pass
    
    def _log_start(self, context: Dict[str, Any]):
        """记录任务开始"""
        self.logger.info(f"{self.name} started", extra={"context": context})
    
    def _log_success(self, result: Dict[str, Any]):
        """记录任务成功"""
        self.logger.info(f"{self.name} completed successfully", extra={"result": result})
    
    def _log_error(self, error: Exception):
        """记录任务失败"""
        self.logger.error(f"{self.name} failed", exc_info=error)
```

from core.agent_base import BaseAgent
from typing import Dict, Any

class ArchitectAgent(BaseAgent):
    def __init__(self):
        super().__init__("ArchitectAgent")
        self.prompt_template = """你是一个资深的软件架构师，请根据用户需求设计系统架构。
        
需求描述：{requirement}

请输出JSON格式的架构设计方案，包含以下字段：
- components: 系统组件列表
- technologies: 技术栈选择
- database: 数据库设计
- api: API接口设计
- deployment: 部署方案"""

    async def execute(self, context: Dict[str, Any]) -> Dict[str, Any]:
        try:
            self._log_start(context)
            
            requirement = context.get("requirement")
            if not requirement:
                raise ValueError("Requirement is missing in context")
            
            prompt = self.prompt_template.format(requirement=requirement)
            result = await context["model_client"].generate(prompt)
            
            if not result["success"]:
                raise Exception(result["error"])
            
            # 解析JSON结果
            import json
            design = json.loads(result["content"])
            
            self._log_success({"design": design})
            return {"success": True, "design": design}
            
        except Exception as e:
            self._log_error(e)
            return {"success": False, "error": str(e)}
```

from core.agent_base import BaseAgent
from typing import Dict, Any
import re

class CoderAgent(BaseAgent):
    def __init__(self):
        super().__init__("CoderAgent")
        self.prompt_template = """你是一个专业的Python开发工程师，请根据架构设计生成代码。
        
架构设计：{design}

请生成可运行的Python代码，要求：
1. 代码要有详细的注释
2. 遵循PEP8规范
3. 包含必要的异常处理
4. 使用类型提示
5. 代码要模块化

请只输出代码内容，不要包含解释说明。"""

    async def execute(self, context: Dict[str, Any]) -> Dict[str, Any]:
        try:
            self._log_start(context)
            
            design = context.get("design")
            if not design:
                raise ValueError("Design is missing in context")
            
            prompt = self.prompt_template.format(design=design)
            result = await context["model_client"].generate(prompt)
            
            if not result["success"]:
                raise Exception(result["error"])
            
            # 提取代码块
            code = self._extract_code(result["content"])
            
            self._log_success({"code": code})
            return {"success": True, "code": code}
            
        except Exception as e:
            self._log_error(e)
            return {"success": False, "error": str(e)}
    
    def _extract_code(self, text: str) -> str:
        """从AI回复中提取代码块"""
        code_blocks = re.findall(r'```python\n(.*?)\n```', text, re.DOTALL)
        if code_blocks:
            return code_blocks[0]
        return text

### 3.4 安全Agent (agents/security.py)

```
from core.agent_base import BaseAgent
from typing import Dict, Any
from config.settings import settings

class SecurityAgent(BaseAgent):
    def __init__(self):
        super().__init__("SecurityAgent")
    
    async def execute(self, context: Dict[str, Any]) -> Dict[str, Any]:
        try:
            self._log_start(context)
            
            # 检查敏感信息
            code = context.get("code", "")
            test_code = context.get("test_code", "")
            
            # 敏感词检查
            for word in settings.SENSITIVE_WORDS:
                if word.lower() in code.lower():
                    raise Exception(f"代码中包含敏感词: {word}")
                if word.lower() in test_code.lower():
                    raise Exception(f"测试代码中包含敏感词: {word}")
            
            # 安全性检查
            security_issues = self._check_security_issues(code)
            if security_issues:
                raise Exception(f"发现安全问题: {', '.join(security_issues)}")
            
            self._log_success({"security_check": "passed"})
            return {"success": True, "message": "Security check passed"}
            
        except Exception as e:
            self._log_error(e)
            return {"success": False, "error": str(e)}
    
    def _check_security_issues(self, code: str) -> list:
        """检查常见的安全问题"""
        issues = []
        
        # 检查SQL注入风险
        if "format(" in code or "%" in code and "sql" in code.lower():
            issues.append("可能存在SQL注入风险")
        
        # 检查硬编码密码
        if "password=" in code.lower() or "secret=" in code.lower():
            issues.append("发现硬编码敏感信息")
        
        # 检查不安全的文件操作
        if "open(" in code and "w" in code and "user_input" in code:
            issues.append("可能存在不安全的文件写入操作")
        
        return issues
```

## 🔄 四、工作流管理

### 4.1 工作流定义 (core/workflow.py)

```
from enum import Enum
from typing import Dict, Any, List
from core.agent_base import BaseAgent

class WorkflowStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"

class WorkflowStep:
    def __init__(self, name: str, agent: BaseAgent, required_keys: List[str] = None):
        self.name = name
        self.agent = agent
        self.required_keys = required_keys or []
    
    async def execute(self, context: Dict[str, Any]) -> Dict[str, Any]:
        # 检查前置条件
        for key in self.required_keys:
            if key not in context:
                return {"success": False, "error": f"Missing required context key: {key}"}
        
        return await self.agent.execute(context)

class Workflow:
    def __init__(self):
        self.steps: List[WorkflowStep] = []
        self.status = WorkflowStatus.PENDING
        self.context: Dict[str, Any] = {}
    
    def add_step(self, step: WorkflowStep):
        self.steps.append(step)
    
    async def run(self) -> Dict[str, Any]:
        try:
            self.status = WorkflowStatus.RUNNING
            
            for step in self.steps:
                result = await step.execute(self.context)
                
                if not result["success"]:
                    self.status = WorkflowStatus.FAILED
                    return {"success": False, "error": result["error"], "step": step.name}
                
                # 合并结果到上下文
                self.context.update(result)
            
            self.status = WorkflowStatus.COMPLETED
            return {"success": True, "context": self.context}
            
        except Exception as e:
            self.status = WorkflowStatus.FAILED
            return {"success": False, "error": str(e)}
```

import logging
import sys
from config.settings import settings

def get_logger(name: str = __name__):
    logger = logging.getLogger(name)
    
    if not logger.handlers:
        handler = logging.StreamHandler(sys.stdout)
        formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        )
        handler.setFormatter(formatter)
        logger.addHandler(handler)
        logger.setLevel(logging.INFO)
    
    return logger
```

### 5.2 监控指标 (utils/metrics.py)

```
from prometheus_client import Counter, Histogram, start_http_server
import time

# 请求计数器
REQUEST_COUNT = Counter(
    'ai_dev_requests_total',
    'Total number of AI development requests',
    ['agent', 'status']
)

# 请求延迟直方图
REQUEST_LATENCY = Histogram(
    'ai_dev_request_duration_seconds',
    'Request latency in seconds',
    ['agent']
)

class Metrics:
    @staticmethod
    def record_request(agent_name: str, success: bool, duration: float):
        status = "success" if success else "failure"
        REQUEST_COUNT.labels(agent=agent_name, status=status).inc()
        REQUEST_LATENCY.labels(agent=agent_name).observe(duration)
    
    @staticmethod
    def start_monitoring(port: int = 8000):
        start_http_server(port)
```

## 🚀 六、应用开发流程

### 6.1 主流程实现 (main.py)

```
import asyncio
import sys
from core.workflow import Workflow, WorkflowStep
from agents.architect import ArchitectAgent
from agents.coder import CoderAgent
from agents.tester import TesterAgent
from agents.security import SecurityAgent
from core.model_client import ModelClient
from utils.metrics import Metrics
from utils.logger import get_logger

logger = get_logger(__name__)

async def develop_application(requirement: str) -> Dict[str, Any]:
    """
    完整的应用开发流程
    """
    # 初始化组件
    model_client = ModelClient()
    workflow = Workflow()
    
    # 添加工作流步骤
    workflow.add_step(WorkflowStep(
        "architecture_design",
        ArchitectAgent(),
        ["requirement", "model_client"]
    ))
    
    workflow.add_step(WorkflowStep(
        "code_generation",
        CoderAgent(),
        ["design", "model_client"]
    ))
    
    workflow.add_step(WorkflowStep(
        "test_generation",
        TesterAgent(),
        ["code", "model_client"]
    ))
    
    workflow.add_step(WorkflowStep(
        "security_check",
        SecurityAgent(),
        ["code", "test_code"]
    ))
    
    # 构建上下文
    context = {
        "requirement": requirement,
        "model_client": model_client
    }
    
    # 执行工作流
    start_time = time.time()
    result = await workflow.run()
    duration = time.time() - start_time
    
    # 记录监控指标
    Metrics.record_request("application_development", result["success"], duration)
    
    return result

def main():
    """命令行入口"""
    if len(sys.argv) < 2:
        print("Usage: python main.py <requirement>")
        sys.exit(1)
    
    requirement = " ".join(sys.argv[1:])
    
    # 启动监控
    Metrics.start_monitoring()
    
    # 执行开发流程
    result = asyncio.run(develop_application(requirement))
    
    if result["success"]:
        print("✅ 开发完成！")
        print(f"代码：\n{result['context']['code']}")
        print(f"测试代码：\n{result['context']['test_code']}")
    else:
        print(f"❌ 开发失败：{result['error']}")

if __name__ == "__main__":
    main()
```
